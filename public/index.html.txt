            <input type="text" id="keyword" placeholder="어떤 알바를 찾으세요?">
            <button onclick="doSearch()">검색</button>
        </div>
        <div id="results"></div>
    </div>

    <script>
        async function doSearch() {
            const kw = document.getElementById('keyword').value;
            const resArea = document.getElementById('results');
            if(!kw) return alert('검색어를 입력해주세요.');

            resArea.innerHTML = "<p style='text-align:center;'>검색 중입니다...</p>";

            try {
                const res = await fetch('/api/search', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ keyword: kw })
                });
                const data = await res.json();
                if(data.error) throw new Error(data.error);

                resArea.innerHTML = data.jobs.map(j => `
                    <div class="job-item">
                        <span class="platform-tag">${j.platform}</span>
                        <a href="${j.link}" target="_blank" class="job-title">${j.title}</a>
                        <p class="snippet">${j.snippet}</p>
                    </div>
                `).join('');
                if(data.jobs.length === 0) resArea.innerHTML = "<p>검색 결과가 없습니다.</p>";
            } catch(e) {
                resArea.innerHTML = "<p style='color:red;'>오류: " + e.message + "</p>";
            }
        }
    </script>
